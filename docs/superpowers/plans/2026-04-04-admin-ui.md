import time
from pathlib import Path
from fastapi import APIRouter
from pydantic import BaseModel
from backend.config import settings, HOT_RELOAD_FIELDS

router = APIRouter(prefix="/api/admin", tags=["admin"])

_start_time: float = 0.0

# Setting field definitions: group → {field → {type, hot_reload}}
_SETTINGS_SCHEMA: dict[str, dict[str, dict]] = {
    "llm": {
        "ollama_host": {"type": "text", "hot_reload": True},
        "ollama_model": {"type": "text", "hot_reload": True},
        "ollama_model_light": {"type": "text", "hot_reload": True},
        "ollama_model_heavy": {"type": "text", "hot_reload": True},
        "ollama_num_ctx": {"type": "number", "hot_reload": True},

# Sensitive fields that should be masked in GET
_SENSITIVE_FIELDS = {"telegram_bot_token"}


def init(start_time: float):
    """Called from main.py lifespan to set server start time."""
    global _start_time
    _start_time = start_time


def write_env_file(env_path: Path, updates: dict[str, str]):
    """Update .env file: change existing keys, append new ones. Preserves comments."""
    if env_path.exists():
        lines = env_path.read_text(encoding="utf-8").splitlines()
    else:
        lines = []

    updated_keys = set()
    new_lines = []
    for line in lines:
        stripped = line.strip()
        if stripped and not stripped.startswith("#") and "=" in stripped:
            key = stripped.split("=", 1)[0].strip()
            if key in updates:
                new_lines.append(f"{key}={updates[key]}")
                updated_keys.add(key)
                continue
        new_lines.append(line)

    # Append new keys
    for key, value in updates.items():
        if key not in updated_keys:
            new_lines.append(f"{key}={value}")

    env_path.write_text("\n".join(new_lines) + "\n", encoding="utf-8")


class SettingsUpdate(BaseModel):
    group: str
    values: dict[str, str]


@router.get("/dashboard")
async def get_dashboard():
    from backend.main import main_agent, budget_tracker, db
    import httpx

    # Active agents
    active = []
    if main_agent:
        active = main_agent.get_status().get("active_agents", [])

    # Open tasks count + recent tasks
    open_count = 0
    recent = []
    if db:
        open_tasks = await db.get_open_tasks()
        open_count = len(open_tasks)
        # Get recent tasks (last 5 by ID desc)
        cursor = await db._conn.execute(
            "SELECT id, title, status, assigned_to FROM tasks ORDER BY id DESC LIMIT 5"
        )
        rows = await cursor.fetchall()
        recent = [
            {"id": r["id"], "title": r["title"], "status": r["status"], "agent": r["assigned_to"] or ""}
            for r in rows
        ]

    # Budget
    budget = {}
    if budget_tracker:
        budget = {
            "used": budget_tracker.used,
            "budget": budget_tracker.daily_budget,
            "remaining": budget_tracker.remaining,
        }

    # Ollama status
    ollama_status = "offline"
    try:
        async with httpx.AsyncClient(timeout=3) as client:
            resp = await client.get(f"{settings.ollama_host}/api/tags")
            if resp.status_code == 200:
                ollama_status = "online"
    except Exception:
        pass

    return {
        "active_agents": active,
        "open_tasks_count": open_count,
        "recent_tasks": recent,
        "budget": budget,
        "uptime_seconds": int(time.time() - _start_time) if _start_time else 0,
        "ollama_status": ollama_status,
    }


@router.get("/settings")
async def get_settings():
    groups = {}
    for group_name, fields in _SETTINGS_SCHEMA.items():
        group_data = {}
        for field_name, field_meta in fields.items():
            raw_value = getattr(settings, field_name, "")
            value = str(raw_value) if raw_value is not None else ""
            if field_name in _SENSITIVE_FIELDS and value:
                value = "***"
            group_data[field_name] = {
                "value": value,
                "hot_reload": field_meta["hot_reload"],
                "type": field_meta["type"],
            }
            if "options" in field_meta:
                group_data[field_name]["options"] = field_meta["options"]
        groups[group_name] = group_data
    return {"groups": groups}


@router.put("/settings")
async def put_settings(update: SettingsUpdate):
    group_fields = _SETTINGS_SCHEMA.get(update.group, {})
    if not group_fields:
        return {"saved": False, "error": f"Unknown group: {update.group}"}

    env_updates = {}
    hot_reloaded = True
    restart_required = False

    for field_name, new_value in update.values.items():
        if field_name not in group_fields:
            continue
        field_meta = group_fields[field_name]
        env_key = field_name.upper()
        env_updates[env_key] = new_value

        if field_meta["hot_reload"] and field_name in HOT_RELOAD_FIELDS:
            # Update in-memory settings
            current_type = type(getattr(settings, field_name, ""))
            if current_type == int:
                setattr(settings, field_name, int(new_value))
            elif current_type == bool:
                setattr(settings, field_name, new_value.lower() in ("true", "1", "yes"))
            elif current_type == Path:
                from pathlib import Path
                setattr(settings, field_name, Path(new_value))
            else:
                setattr(settings, field_name, new_value)
        else:
            hot_reloaded = False
            restart_required = True

    # Write to .env
    env_path = Path(".env")
    write_env_file(env_path, env_updates)

    return {
        "saved": True,
        "hot_reloaded": hot_reloaded,
        "restart_required": restart_required,
    }

                const isServer = groupName === 'server';
                const label = GROUP_LABELS[groupName] || groupName;
                const badge = isServer ? '<span class="badge">Neustart nötig</span>' : '';

                let fieldsHtml = '';
                for (const [fieldName, meta] of Object.entries(fields)) {
                    let inputHtml;
                    if (meta.type === 'password') {
                        inputHtml = `
                            <input type="password" id="f-${fieldName}" value="${meta.value === '***' ? '' : meta.value}"
                                   placeholder="${meta.value === '***' ? '(gesetzt)' : '(leer)'}">
                            <button class="pw-toggle" onclick="togglePw('f-${fieldName}')">👁</button>
                        `;
                    } else if (meta.type === 'boolean') {
                        const checked = meta.value === 'True' || meta.value === 'true' ? 'checked' : '';
                        inputHtml = `<input type="checkbox" id="f-${fieldName}" ${checked} style="width:auto">`;
                    } else if (meta.type === 'select' && meta.options) {
                        const opts = meta.options.map(o =>
                            `<option value="${o}" ${o === meta.value ? 'selected' : ''}>${o}</option>`
                        ).join('');
                        inputHtml = `<select id="f-${fieldName}">${opts}</select>`;
                    } else {
                        const inputType = meta.type === 'number' ? 'number' : 'text';
                        inputHtml = `<input type="${inputType}" id="f-${fieldName}" value="${meta.value}">`;
                    }
                    fieldsHtml += `
                        <div class="field-row">
                            <label for="f-${fieldName}">${fieldName}</label>
                            ${inputHtml}
                        </div>
                    `;
                }

                container.innerHTML += `
                    <div class="settings-group" data-group="${groupName}">
                        <h3>${label} ${badge}</h3>
                        ${fieldsHtml}
                        <div class="group-footer">
                            <button class="save-btn" onclick="saveGroup('${groupName}')">Speichern</button>
                            <span class="save-msg" id="msg-${groupName}"></span>
                        </div>
                    </div>
                `;
            }
        } catch (e) {
            console.error('Settings load error:', e);
        }
    }

    async function saveGroup(groupName) {
        const group = document.querySelector(`.settings-group[data-group="${groupName}"]`);
        const inputs = group.querySelectorAll('input, select');
        const values = {};
        inputs.forEach(el => {
            const name = el.id.replace('f-', '');
            if (el.type === 'checkbox') {
                values[name] = el.checked ? 'true' : 'false';
            } else if (el.type === 'password' && el.value === '') {
                // Don't send empty password fields (keep existing)
                return;
            } else {
                values[name] = el.value;
            }
        });

        try {
            const resp = await fetch('/api/admin/settings', {
                method: 'PUT',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({group: groupName, values}),
            });
            const data = await resp.json();
            const msgEl = document.getElementById(`msg-${groupName}`);
            if (data.saved) {
                if (data.restart_required) {
                    msgEl.className = 'save-msg warn';
                    msgEl.textContent = '⚠ Gespeichert — Neustart nötig';
                } else {
                    msgEl.className = 'save-msg ok';
                    msgEl.textContent = '✓ Gespeichert & aktiv';
                }
            } else {
                msgEl.className = 'save-msg warn';
                msgEl.textContent = '✗ Fehler: ' + (data.error || 'unbekannt');
            }
            setTimeout(() => { msgEl.textContent = ''; }, 5000);
        } catch (e) {
            console.error('Save error:', e);
        }
    }

    function togglePw(id) {
        const el = document.getElementById(id);
        el.type = el.type === 'password' ? 'text' : 'password';
    }

    loadSettings();
    setInterval(loadDashboard, 10000);
